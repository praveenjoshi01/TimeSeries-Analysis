### Decomposition of time series into components

# Subset the data to have required columns only
GE.sub <- GE %>% select(Date, Close)

# Converting the data into time series object
GE.zoo <- zoo(GE.sub$Close, order.by = GE.sub$Date)
GE.ts <- ts(GE.zoo,frequency = 12, start = c(2011,1))

#Converting test dataset to timeseries
# Subset the data to have required columns only
GE.subTest <- GE.test %>% select(Date, Close)

# Converting the data into time series object
GE.zooTest <- zoo(GE.subTest$Close, order.by = GE.subTest$Date)
GE.tsTest <- ts(GE.zooTest,frequency = 12, start = c(2015,1))

# Decomposing the ts into components
GE.timeSeriesComponent <- decompose(GE.ts)
